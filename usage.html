1. Convert base64 ↔ File

import { fileToBase64, base64ToFile } from 'fileasy';

async function roundtrip(file) {
  const txt = await fileToBase64(file);
  return base64ToFile(txt, `copy.${file.type.split('/')[1]}`, file.type);
}

2. Image format conversion

import { pngToJpeg } from 'fileasy';

async function convertPngToJpg(pngFile) {
  const jpg = await pngToJpeg(pngFile, 'converted.jpg');
